\DeclareCiteCommand{\templateshortcite}
  {}
  {%
    \iffieldundef{doi}
      {\iffieldundef{url}
        {\printtext{%
          \printnames{labelname}%
          \setunit{\addspace}%
          \printfield{year}%
        }}
        {\href{\thefield{url}}{\textcolor{mygreencite}{%
          \printnames{labelname}%
          \setunit{\addspace}%
          \printfield{year}%
        }}}}
      {\href{https://doi.org/\thefield{doi}}{\textcolor{mygreencite}{%
        \printnames{labelname}%
        \setunit{\addspace}%
        \printfield{year}%
      }}}%
  }
  {\multicitedelim}
  {}

\newcommand{\templatecustomcitetext}{}
\DeclareCiteCommand{\templatecustomcite}
  {}
  {%
    \iffieldundef{doi}
      {\iffieldundef{url}
        {\printtext{\templatecustomcitetext}}
        {\href{\thefield{url}}{\textcolor{mygreencite}{\templatecustomcitetext}}}}
      {\href{https://doi.org/\thefield{doi}}{\textcolor{mygreencite}{\templatecustomcitetext}}}%
  }
  {\multicitedelim}
  {}

\DeclareCiteCommand{\templateplaincite}
  {}
  {%
    \printtext{%
      \printnames{labelname}%
      \setunit{\addspace}%
      \printfield{year}%
    }%
  }
  {\multicitedelim}
  {}
